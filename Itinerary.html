    <script>

        function slideImages(sliderId, direction) {

            const slider = document.getElementById(sliderId);

            const images = slider.querySelector('.slider-images');

            const imageWidth = slider.offsetWidth;

            const currentTransform = parseFloat(images.style.transform.replace('translateX(', '').replace('px)', '')) || 0;

            const maxScroll = -(images.children.length - 1) * imageWidth;


            if (direction === 'left' && currentTransform < 0) {

                images.style.transform = `translateX(${currentTransform + imageWidth}px)`;

            } else if (direction === 'right' && currentTransform > maxScroll) {

                images.style.transform = `translateX(${currentTransform - imageWidth}px)`;

            }

        }

    </script>
